\begin{exercicioBanco}[\valorquestao]
Considere a circunferência de equação \(C_{1}: x^2 + y^2 - 2x - 4y + 1 = 0\) e a reta \(r: y = 4\). Classifique a posição relativa entre a reta e a circunferência.

% Define as alternativas
\newcommand{\alternativas}{%
    \begin{center}
        \begin{tabularx}{\textwidth}{XXXXX}
            (a) Secante. &
            (b) Tangente. &
            (c) Externa. &
            (d) Coincidente. &
            (e) Paralela.
        \end{tabularx}
    \end{center}
}

\end{exercicioBanco}

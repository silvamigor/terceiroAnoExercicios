% Define as alternativas
\newcommand{\alternativas}{%

        \begin{tabularx}{\textwidth}{XXXXX}

            (b) \(\n{128}\). &
            (c) \(\n{127}\). &
            (d) \(\n{42}\). &
            (e) \(\n{21}\).
        \end{tabularx}
    \end{center}
}
